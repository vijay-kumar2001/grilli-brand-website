# üß© 1. START / SETUP
git clone <repo_url>                        # First time only: clone full project to your machine
git config --global user.name "Your Name"   # Set your name for all repos (one-time setup)
git config --global user.email "your@email.com"  # Set your GitHub email (one-time setup)

# üß† 2. BEFORE STARTING WORK (Always Do This First)
git fetch origin --prune                    # Update all remote branch info and remove deleted ones
git switch main                             # Move to main branch
git pull --ff-only origin main              # Safely update local main to latest remote main

# üå± 3. CREATE A NEW BRANCH FOR A FEATURE
git switch -c feature/short-descriptor      # Create & switch to new feature branch from main
# Example: git switch -c feature/responsive-navbar

# üíª 4. WORK ON THE FEATURE
# (edit files normally)
git add .                                   # Stage all modified/new files
git commit -m "feat: short, clear message"  # Commit your work with short, focused message
# Example: "feat: add responsive navbar for desktop"

# ‚òÅÔ∏è 5. PUSH THE BRANCH TO GITHUB
git push -u origin feature/short-descriptor # Push branch to GitHub and set upstream (run once per new branch)

# üîÅ 6. CONTINUE WORK LATER / ON ANOTHER MACHINE
git fetch origin                            # Update remote branch info
git switch -c feature/short-descriptor origin/feature/short-descriptor  # First time only if branch not local
# OR if already exists locally:
git switch feature/short-descriptor         # Switch to existing local branch
git pull                                    # Get latest commits from GitHub

# üß© 7. KEEP YOUR BRANCH UPDATED WITH MAIN
git fetch origin                            # Get latest remote changes
git switch feature/short-descriptor         # Ensure you're on your feature branch
git rebase origin/main                      # Reapply your commits on top of latest main
# OR (simpler, keeps merge history)
git merge origin/main                       # Merge latest main into your branch

# Resolve conflicts if any:
git add <fixed_files>                       # Mark resolved files as fixed
git rebase --continue                       # Continue rebase after fixing conflicts
# or if merged:
git commit                                  # Complete merge commit

# üöÄ 8. PUSH AFTER REBASE OR REGULAR WORK
git push origin feature/short-descriptor    # Normal push (no rebase)
# If rebased, force safely:
git push --force-with-lease origin feature/short-descriptor  # Safe force-push after rebase

# üßë‚Äçüíª 9. CREATE PULL REQUEST (PR) ON GITHUB (manual)
# Go to GitHub ‚Üí ‚ÄúCompare & pull request‚Äù
# Write a clear title & description, submit PR when feature is complete
# Merge pull request ‚Üí Confirm merge

# üßπ 10. AFTER MERGE ‚Äî CLEAN UP
git fetch origin --prune                    # Update branch info
git switch main                             # Go back to main
git pull --ff-only origin main              # Get merged work into local main
git branch -d feature/short-descriptor      # Delete merged local branch
git push origin --delete feature/short-descriptor  # Delete branch from GitHub

# üîÅ 11. SWITCHING BETWEEN FEATURES (Optional)
git stash push -m "WIP: navbar"             # Save uncommitted work temporarily
git switch feature/other-branch             # Move to another branch
git stash pop                               # Reapply your saved work

# üì° 12. UPDATE AFTER SOMEONE ELSE‚ÄôS MERGE / ANOTHER DEVICE‚ÄôS WORK
git fetch origin --prune                    # Update remote refs
git switch main                             # Move to main
git pull --ff-only origin main              # Update local main safely

# üîç 13. USEFUL CHECKS
git status                                  # See what branch you‚Äôre on and what‚Äôs modified
git branch -vv                              # See local branches + their remote tracking
git branch -r                               # List all remote branches
git log --oneline --graph --decorate --all | head -20  # See recent commit graph

# üß† 14. GOLDEN RULES
# ‚úÖ One branch per feature
# ‚úÖ Push frequently (safe backups + visible progress)
# ‚úÖ Only merge to main when feature is complete
# ‚úÖ Always pull main before creating new branches
# ‚úÖ Clean up merged branches regularly
# ‚úÖ Rebase/merge your feature with main before merging PR

# ‚ö° TL;DR ‚Äî DAILY ROUTINE SUMMARY
git fetch origin --prune                    # Update remote branches
git switch main                             # Switch to main
git pull --ff-only origin main              # Update main to latest
git switch -c feature/<new-feature>         # Create new or switch existing
# work, commit, push, rebase/merge main, create PR ‚Üí merge ‚Üí cleanup
